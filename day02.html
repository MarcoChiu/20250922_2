Remove String Spaces - 移除字串中的空格 

        function noSpace(x) {
            return x.split("").filter(e => {
                return e !== " "
            }).join("");
        }
        console.log(noSpace("8 j 8   mBliB8g  imjB8B8  jl  B"));
