        function wordsToMarks(string) {
            let sum = 0;
            const arr = ['a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k', 'l', 'm', 'n', 'o', 'p', 'q', 'r', 's', 't', 'u', 'v', 'w', 'x', 'y', 'z'];
            const arrObj = [];
            arr.forEach((v, i) => {
                arrObj.push({ [v]: i + 1 });
            });

            string.split('').forEach(v => {
                const obj = arrObj.find(o => o[v]);
                sum += obj[v];
            });
            return sum;
        }
