Math

        //Math.floor(x)，回傳不大於 x 的最大整數值，無條件捨去。
        console.log(Math.floor(3.44));

        //Math.ceil(x)，回傳不小於 x 的最小整數值，無條件進位。
        console.log(Math.ceil(3.44));

        //Math.pow(x, y)，回傳 x 的 y 次方，也就是 x^y，n次方。
        console.log(Math.pow(2, 3));

        //Math.random()，回傳一個 0 到 1 之間的偽隨機值，亂數。
        console.log(Math.random());

        //Math.round(x)，回傳 x 的四捨五入值。
         console.log(Math.round(3.44));
         console.log(Math.round(3.5));

        //Math.min([x[, y[, …]]])，回傳給定數值中的最小值。
        //Math.max([x[, y[, …]]])，回傳給定數值中的最大值。
        //直接給值
        console.log(Math.max(1, 5, 8, 2));
        //陣列
        const array = [1, 3, 2];
        console.log(Math.max(...array));
        //物件
        const users = [{ name: 'Alice', age: 30 }, { name: 'Bob', age: 25 }, { name: 'Charlie', age: 35 }];
        console.log(Math.max(...users.map(user => user.age)));
