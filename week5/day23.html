        function rowSumOddNumbers(n) {
            var start = n * n - n + 1;
            let i = 1;
            let c = [];
            while (c.length < n) {

                if (i % 2 != 0) {
                    if (i >= start) {
                        c.push(i);
                    }
                }
                i++;
            }
            return c.reduce((a, b) => a + b, 0);
        }
