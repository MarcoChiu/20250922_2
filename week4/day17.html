        function likeOrDislike(buttons) {
            let state = 'Nothing';
            buttons.forEach(function(item,i) {
                if (buttons[i] === state) {
                    state = 'Nothing';
                } else {
                    state = buttons[i];
                }
            });       
            return state;
        }
