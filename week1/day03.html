day3-清掃字串 

        function stringClean(s) {
            return s.split("").filter(e => {
                return !isNaN(parseInt(e)) === false;
            }).join("");
        }

        // 測試範例
        console.log(stringClean('! !')); // '!!'
        console.log(stringClean('123456789'));//-> ''
        console.log(stringClean('This looks5 grea8t!'));// 'This looks great!'
